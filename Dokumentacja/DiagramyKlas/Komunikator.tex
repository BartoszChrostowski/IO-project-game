\subsubsection{Klasa Communicator}
\begin{itemize}
    \method[private Socket]{gmSocket}{Gniazdo do połączeń TCP z Game Masterem.}
    \method[private Socket]{agentSocket}{Gniazdo do połączeń TCP z Agentami.}
\end{itemize}
\paragraph{Metody}
\begin{itemize}
    \method{SendMessageToAgent()}{Metoda wysyłająca odpowiednią wiadomość wiadomość do Agenta za pomocą odpowiedniego gniazda.}
    \method{SendMessageToGM()}{Metoda wysyłająca odpowiednią wiadomość wiadomość do Game Mastera za pomocą odpowiedniego gniazda. Dodaje identyfikator agenta. }
\end{itemize}

\subsubsection{Klasa AgentMessage}
\begin{itemize}
    \method[public int]{id}{Identyfikator agenta do którego przesyłana jest wiadomość.}
	\method[public string]{payload}{Treść wiadomości przesyłana od lub do agenta w formacie JSON.}
\end{itemize}

\subsubsection{Klasa GMMessage}
\begin{itemize}
	\method[public string]{payload}{Treść wiadomości przesyłana od lub do Game Mastera w formacie JSON.}
\end{itemize}
